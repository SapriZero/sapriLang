//! Macro pure e leggere per il layer base.
//! Zero dipendenze esterne. Espansione immediata.

pub mod eval;
pub mod fp;
pub mod error;

// Re-export per accesso modulare (es. crate::macros::eval)
pub use eval::{eval, mask};
pub use fp::{curry, lazy_if};
pub use error::{try_or, unwrap_or_else};

--$FILE===: core/base/src/macros/eval.rs
//! Valutazione condizionale pura e masking leggero.

/// Valuta un'espressione solo se la condizione è vera, altrimenti fallback.
#[macro_export]
macro_rules! eval {
    ($cond:expr, $then:expr, $else:expr) => {
        if $cond { $then } else { $else }
    };
    ($cond:expr, $then:expr) => {
        if $cond { Some($then) } else { None }
    };
}

/// Applica una maschera/trasformazione a un valore solo se soddisfa un predicato.
#[macro_export]
macro_rules! mask {
    ($val:expr, $pred:expr, $transform:expr) => {
        if $pred(&$val) { $transform($val) } else { $val }
    };
}

--$FILE===: core/base/src/macros/fp.rs
//! Utility funzionali leggere.

/// Crea una closure curry da una funzione a 2 argomenti.
#[macro_export]
macro_rules! curry {
    ($fn:path) => {
        |a| move |b| $fn(a, b)
    };
}

/// Esegue il blocco solo se la condizione è vera, altrimenti ritorna None.
#[macro_export]
macro_rules! lazy_if {
    ($cond:expr, $block:block) => {
        if $cond { Some($block) } else { None }
    };
}

--$FILE===: core/base/src/macros/error.rs
//! Macro ergonomiche per gestione errori nel layer base.

/// Tenta un'operazione, se fallisce ritorna un valore di fallback.
#[macro_export]
macro_rules! try_or {
    ($expr:expr, $fallback:expr) => {
        match $expr {
            Ok(v) => v,
            Err(_) => $fallback,
        }
    };
}

/// Variante di `unwrap_or_else` per macro, evita borrow checker su closure complesse.
#[macro_export]
macro_rules! unwrap_or_else {
    ($opt:expr, $default:expr) => {
        match $opt {
            Some(v) => v,
            None => $default,
        }
    };
}
